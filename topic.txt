# Functions for frequently used commands
function klogs() {
  kubectl logs "$1" --tail=100 -f
}

function kexec() {
  kubectl exec -it "$1" -- /bin/bash
}

function kswitch() {
  kubectl config use-context "$1"
  kubectl config set-context --current --namespace="$2"
}
