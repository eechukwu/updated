provider "aws" {
  region = "us-west-2" # Adjust as necessary
}

# Define the API Gateway
resource "aws_api_gateway_rest_api" "apikey1" {
  name        = "apikey1-api"
  description = "API Gateway for apikey1"
}

# Create an API key
resource "aws_api_gateway_api_key" "apikey1" {
  name        = "apikey1"
  description = "API key for apikey1"
  enabled     = true
}

# Create a usage plan
resource "aws_api_gateway_usage_plan" "apikey1" {
  name        = "apikey1-usage-plan"
  description = "Usage plan for apikey1"
}

# Associate the API key with the usage plan
resource "aws_api_gateway_usage_plan_key" "apikey1" {
  key_id        = aws_api_gateway_api_key.apikey1.id
  key_type      = "API_KEY"
  usage_plan_id = aws_api_gateway_usage_plan.apikey1.id
}

# Define a resource in the API
resource "aws_api_gateway_resource" "example_resource" {
  rest_api_id = aws_api_gateway_rest_api.apikey1.id
  parent_id   = aws_api_gateway_rest_api.apikey1.root_resource_id
  path_part   = "example"
}

# Define a method for the resource
resource "aws_api_gateway_method" "example_method" {
  rest_api_id   = aws_api_gateway_rest_api.apikey1.id
  resource_id   = aws_api_gateway_resource.example_resource.id
  http_method   = "GET"
  authorization = "NONE"
  api_key_required = true
}

# Define the integration for the method
resource "aws_api_gateway_integration" "example_integration" {
  rest_api_id             = aws_api_gateway_rest_api.apikey1.id
  resource_id             = aws_api_gateway_resource.example_resource.id
  http_method             = aws_api_gateway_method.example_method.http_method
  type                    = "MOCK"
  request_templates = {
    "application/json" = "{\"statusCode\": 200}"
  }
}

# Deploy the API
resource "aws_api_gateway_deployment" "apikey1" {
  depends_on = [
    aws_api_gateway_integration.example_integration
  ]
  rest_api_id = aws_api_gateway_rest_api.apikey1.id
  stage_name  = "prod"
}

	•	Defining a resource and method that require the API key.
	•	Setting up a mock integration for the method.
	•	Deploying the API.
